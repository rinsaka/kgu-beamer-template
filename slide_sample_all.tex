\section{数式，図，表}

\begin{frame}{数式}
  \begin{itemize}
    \item 数式$F(x)$は\LaTeX~の書き方がほぼ使えるはず
    \begin{equation}
      F(x) = \int_{0}^{x}f(t)dt
    \end{equation}
    \item 数式番号が不要であれば
    \begin{equation*}
      F(x) = \int_{0}^{x}f(t)dt
    \end{equation*}
    \item 複数行の数式ももちろん可能
    \begin{eqnarray*}
      F(x) &=& \int_{0}^{x}f(t)dt \\
      &=& 1 - e^{-\lambda x}
    \end{eqnarray*}
  \end{itemize}
\end{frame}
